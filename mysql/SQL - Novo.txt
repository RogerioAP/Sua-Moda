drop database if exists suamoda;
create database suamoda;
use suamoda;

CREATE TABLE imagens (
idProduto int,
imagem1 varchar(100),
imagem2 varchar(100),
imagem3 varchar(100)
)engine=innodb;

CREATE TABLE compra (
idcompra int PRIMARY KEY,
CPF varchar(100),
data varchar(100),
idProduto int
)engine=innodb;

CREATE TABLE usuario (
CPF varchar(100) PRIMARY KEY,
Nome varchar(100),
Senha varchar(100),
Email varchar(100),
Sobrenome varchar(100),
Telefone varchar(100),
Cidade varchar(100),
Foto varchar(100),
Estilo varchar(100)
)engine=innodb;

CREATE TABLE Endereco (
rua varchar(100),
bairro varchar(100),
numero varchar(100)
)engine=innodb;

CREATE TABLE Administrador (
CPF varchar(100) PRIMARY KEY,
Nome varchar(100),
Senha varchar(100),
Email varchar(100)
)engine=innodb;

CREATE TABLE Produto (
idProduto int PRIMARY KEY,
Nome varchar(100),
Descricao varchar(100),
Categoria varchar(100),
Preco float
)engine=innodb;

ALTER TABLE compra ADD FOREIGN KEY(CPF) REFERENCES Usuario (CPF);
ALTER TABLE compra ADD FOREIGN KEY(idProduto) REFERENCES Produto (idProduto);
ALTER TABLE imagens ADD FOREIGN KEY(idProduto) REFERENCES Produto (idProduto);
